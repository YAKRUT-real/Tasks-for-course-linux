16. Проверка и включение/выключение вручную
#!/bin/bash
free -h
echo "Доступно RAM: $(free -h | awk '/Mem:/{print $7}')"
read -p "Отключить swap? (y/N) " answer
if [ "$answer" = "y" ]; then
    sudo swapoff -a && echo "Swap отключён" && free -h
    read -p "Включить обратно? (y/N) " ans2
    [ "$ans2" = "y" ] && sudo swapon -a && echo "Swap включён" && free -h
fi
